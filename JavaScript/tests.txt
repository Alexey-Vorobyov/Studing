function getDecimal(num) {
var a = ((num) - Math.floor(num)),
b = (Math.ceil(num) - (num));
if (num > 0) {
return (Math.round(a * 1000) / 1000);
} else {
return (Math.round(b * 1000) / 1000);
}
}

alert( getDecimal(12.345) ); // 0.345
